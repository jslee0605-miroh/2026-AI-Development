\begin{frame}{What you will build today}
\begin{itemize}
  \item A tiny tool registry (MCP-like): names, descriptions, input schemas
  \item A minimal orchestrator loop: plan $\rightarrow$ tool calls $\rightarrow$ verify
  \item Guardrails: allow-lists, step limits, and basic injection resistance
  \item A research brief deliverable grounded in local docs + a CSV
\end{itemize}
\end{frame}

\begin{frame}{Business problem}
\textbf{Scenario}: An analyst needs a research brief from messy internal docs + a dataset.
\vspace{0.4cm}

\textbf{Goal}: Produce a structured brief with:
\begin{itemize}
  \item key findings (grounded in sources)
  \item risks and assumptions
  \item recommended next steps
  \item a small table from the dataset (Python tool)
\end{itemize}
\end{frame}

\begin{frame}{Inputs (provided in \texttt{lecture\_4/data})}
\begin{itemize}
  \item \texttt{docs/}: internal docs (including a malicious prompt-injection example)
  \item \texttt{market\_signals.csv}: small dataset to summarize
  \item \texttt{tool\_registry.json}: tool definitions (names + schemas)
\end{itemize}
\end{frame}

\begin{frame}{New workflow primitive introduced}
\begin{itemize}
  \item \textbf{Tool use}: model chooses among tools, but only within constraints
  \item \textbf{Orchestration}: plan/execute/verify with a step budget
  \item \textbf{Defense}: treat retrieved text as untrusted input
\end{itemize}
\end{frame}

\begin{frame}{Deliverable}
\begin{itemize}
  \item Notebook: \texttt{notebooks/lecture\_4\_tooling\_orchestrator.ipynb}
  \item Output: \texttt{data/outputs/research\_brief.md}
\end{itemize}
\end{frame}

\begin{frame}{Extensions / Optional challenges}
\begin{itemize}
  \item \textbf{Verification step}: add a checker that validates claims against tool trace + sources
  \item \textbf{Planning robustness}: re-plan on tool errors; add stopping conditions and max-step budgets
  \item \textbf{Strict tool schemas}: validate tool-call JSON against schemas; reject malformed calls
  \item \textbf{Security hardening}: demonstrate injection and add filters (treat docs as data; never execute doc instructions)
  \item \textbf{Swap-in real MCP}: replace the in-notebook tool registry with an MCP server later
\end{itemize}
\end{frame}
